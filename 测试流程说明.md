### 脚本开发说明

自动化平台通过调用系统cmd或shell命令执行脚本，下面以信而泰网络测试仪表脚本开发为例：

#### 脚本预设文件param.json

平台在创建测试套件过程中支持更具实际物理拓扑和其他测试环境生成脚本预设文件param.json文件，以下边这个文件为例子：
```json
{
    "_comment": "将脚本使用到的变量放到param中",
    "param": {
        "product": "DARYU",
        "time": 10,
        "port": [
            "//10.0.11.191/1/15",
            "//10.0.11.191/1/16"
        ]
    }
}
```
脚本中的需要使用的变量我们都放到了param这个字典中：
product：表示仪表产品类型
time：表示网络测试仪表流量发送时长
port：表示本次脚本执行实际要使用的网络仪表物理端口地址

#### 脚本文件
网络测试仪表支持python语言编写脚本，以下是脚本源码，脚本中读取param.json获取预设的：product、time以及port参数：

```python
import json
import os
import sys

os.environ['RENIX_SERVER_PATH'] = r'C:\Program Files\Xinertel\Renix3.2.8\server'
sys.path.insert(0, os.path.join(os.environ['RENIX_SERVER_PATH'], 'python_api'))

from renix_py_api.renix import *
# 保存配置文件
dirname, tempfilename = os.path.split(os.path.abspath(__file__))
filename, extension = os.path.splitext(tempfilename)
with open(os.path.join(dirname, 'param.json'), "r", encoding="utf-8") as fid:
    data = json.load(fid)

verdict = 'pass'
errInfo = ''
result_data = False

try:

    initialize()

    # 初始化之后，根节点就产生了，可以修改根节点的产品类型
    sys_entry = get_sys_entry()

    sys_entry.edit(ProductType=data['param']['product'])

    # Reserve port1 and port2 in card1 of chassis 192.168.10.1

    sys_entry = get_sys_entry()

    port_location = data['param']['port']

    port1 = Port(upper=sys_entry, Location=port_location[0])

    port2 = Port(upper=sys_entry, Location=port_location[1])

    bring_port_online_cmd = BringPortsOnlineCommand(PortList=[port1.handle, port2.handle])

    bring_port_online_cmd.execute()

    # Create StreamTemplates on Port_1

    stream1 = StreamTemplate(upper=port1)

    stream2 = StreamTemplate(upper=port2)

    # Specify Port_2 as the receiving port of StreamTemplate_1

    stream1.set_relatives('RxPort', port2, EnumRelationDirection.TARGET)

    stream2.set_relatives('RxPort', port1, EnumRelationDirection.TARGET)

    # Configure to get the StreamTemplate's statistics

    resultview_create = CreateResultViewCommand(DataClassName=StreamBlockStats.cls_name())

    resultview_create.execute()

    page_result_view = sys_entry.get_children('PageResultView')[0]

    # Subscribe the statistics

    subscribe_result_cmd = SubscribeResultCommand(ResultViewHandles=page_result_view.handle)

    subscribe_result_cmd.execute()

    # Commit the above configuration

    commit = CommitCommand()

    commit.execute()

    # Start StreamTemplate_1 and wait for 6000 milliseconds

    startallstream = StartAllStreamCommand()

    startallstream.execute()

    time.sleep(data['param']['time'])

    # Stop StreamTemplate_1

    stopallstream = StopAllStreamCommand()

    stopallstream.execute()

    time.sleep(3)

    # View the StreamTemplate's statistics

    result_query = page_result_view.get_children()[0]

    streamblock1_stats = result_query.get_children('StreamBlockStats')[0]

    print(streamblock1_stats.__dict__)

    streamblock2_stats = result_query.get_children('StreamBlockStats')[1]

    print(streamblock1_stats.__dict__)
    result_data = {
        streamblock1_stats.StreamBlockID: {
            "TxPortID": streamblock1_stats.TxPortID,
            "RxPortID": streamblock1_stats.RxPortID,
            "TxStreamFrames": streamblock1_stats.TxStreamFrames,
            "RxStreamFrames": streamblock1_stats.RxStreamFrames
        },
        streamblock2_stats.StreamBlockID: {
            "TxPortID": streamblock2_stats.TxPortID,
            "RxPortID": streamblock2_stats.RxPortID,
            "TxStreamFrames": streamblock2_stats.TxStreamFrames,
            "RxStreamFrames": streamblock2_stats.RxStreamFrames
        }
    }


except Exception as e:
    verdict = 'fail'
    errInfo = repr(e)
finally:
    result = {
        "_comment": "将脚本生成的数据放到result中",
        "result": {
            "errInfo": errInfo,
            "verdict": verdict,
            "data": result_data
        }
    }
    with open(os.path.join(dirname, 'result.json'), "w", encoding="utf-8") as f:
        json.dump(result, f, indent=4, ensure_ascii=False)
    print(f'errInfo:\n{errInfo}')
    print(f'verdict:{verdict}')

```

#### 脚本结果和数据保存到result.json文件中

从上诉脚本可以看出脚本中将测试的测试结果以及本测试所关注的关键性测试数据存放到了result.json文件中，平台通过读取这个文件分析脚本执行结果和人关键测试数据。
errInfo：脚本如果失败，存放脚本错误信息
verdict：脚本成功为：pass，脚本失败为：fail
data：测试关键数据
```json
{
    "_comment": "将脚本生成的数据放到result中",
    "result": {
        "errInfo": "",
        "verdict": "pass",
        "data": {
            "StreamTemplate_1": {
                "TxPortID": "Port_1",
                "RxPortID": "Port_2",
                "TxStreamFrames": 7239401,
                "RxStreamFrames": 7239401
            },
            "StreamTemplate_2": {
                "TxPortID": "Port_2",
                "RxPortID": "Port_1",
                "TxStreamFrames": 7271455,
                "RxStreamFrames": 7271455
            }
        }
    }
}
```
